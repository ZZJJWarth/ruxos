#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

int main()
{
     // 尝试访问内核空间内存地址
    volatile uint64_t *kernel_address = (uint64_t *)0xffff000040080000;

    // 尝试读取内核地址中的值（这个行为在正常的操作系统中应当引发段错误）
    uint64_t value = *kernel_address;

    printf("Accessed kernel memory, value: %lu\n", value);
    printf("Hello, %c app!\n", 'C');
    return 0;
}
